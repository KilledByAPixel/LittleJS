Example
let a = new Color;              // white
let b = new Color(1, 0, 0);     // red
let c = new Color(0, 0, 0, 0);  // transparent black
let d = rgb(0, 0, 1);           // blue using rgb color
let e = hsl(.3, 1, .5);         // green using hsl color
